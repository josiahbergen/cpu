
0   LOAD    reg,                imm16 | reg:reg     load word                       reg <- [imm16 | reg:reg]
1   STORE   imm16 | reg:reg     reg                 store word                      [imm16 | reg:reg] <- reg
2   MOVE    reg,                reg | imm8          move word                       reg <- reg | imm8

3   PUSH    reg | imm8                              push to stack                   [SP--] <- imm8/reg   *note that the stack grows downwards
4   POP     reg                                     pop from stack                  reg <- [++SP]        *note that the stack grows downwards

5   ADD     reg,                reg | imm8          add                             reg <- reg + (imm8 | reg) 
6   ADDC    reg,                reg | imm8          add with carry                  reg <- reg + (imm8 | reg) + c

7   SUB     reg,                reg | imm8          subtract                        reg <- reg - (imm8 | reg)
8   SUBB    reg,                reg | imm8          subract with borrow             reg <- reg - (imm8 | reg) - b

9   INC     reg                                     increment                       reg <- reg + 1
10  DEC     reg                                     decrement                       reg <- reg - 1

11  SHL     reg,                reg | imm8          bit shift left                  reg <- reg << (reg | imm8)
12  SHR     reg,                reg | imm8          bit shift right                 reg <- reg >> (reg | imm8)

13  AND     reg,                reg | imm8          bitwise and                     reg <- reg AND (reg | imm8)
14  OR      reg,                reg | imm8          bitwise or                      reg <- reg OR  (reg | imm8)
15  NOT     reg,                reg | imm8          bitwise not                     reg <- reg NOT (reg | imm8)
16  NOR     reg,                reg | imm8          bitwise nor                     reg <- reg NOR (reg | imm8)
17  XOR     reg,                reg | imm8          bitwise xor                     reg <- reg XOR (reg | imm8)

18  INB     reg,                port(reg | imm8)    get byte from I/O port          reg <- port(reg | imm8)
19  OUTB    reg,                port(reg | imm8)    send byte through I/O port      port(reg | imm8) <- reg

20  CALL    imm16 | reg:reg                         call subroutine                 this istruction is a built-in macro
21  RET                                             return from subroutine          this istruction is a built-in macro

22  CMP     reg,                reg | imm8          compare
23  SEC                                             set carry flag                  carry flag <- 1
24  CLC                                             clear carry flag                carry flag <- 0
25  CLZ                                             clear zero flag                 zero flag <- 0

26  JMP     imm16 | reg:reg                         unconditional jump              PC <- [imm16 | reg:reg]
27  JNZ     imm16 | reg:reg                         jump of not zero                PC <- [imm16 | reg:reg] if zero flag is 1 else PC
28  JZR     imm16 | reg:reg                         jump if zero                    PC <- [imm16 | reg:reg] if zero flag is 0 else PC
29  JC      imm16 | reg:reg                         jump if carry                   PC <- [imm16 | reg:reg] if carry flag is 1 else PC
30  JNC     imm16 | reg:reg                         jump if not carry               PC <- [imm16 | reg:reg] if carry flag is 0 else PC

31  HLT                                             halt                            halt flag <- 1
32  NOP                                             no operation                    n/a
